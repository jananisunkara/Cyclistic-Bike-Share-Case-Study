4. Rename 2019 Columns

q1_2019 <- q1_2019 %>%
  rename(
    ride_id = trip_id,
    started_at = start_time,
    ended_at = end_time,
    start_station_name = from_station_name,
    start_station_id = from_station_id,
    end_station_name = to_station_name,
    end_station_id = to_station_id,
    member_casual = usertype
  ) %>%
  mutate(
    rideable_type = NA,
    start_lat = NA,
    start_lng = NA,
    end_lat = NA,
    end_lng = NA
  )

5. Fix Datatype Mismatches

# Convert timestamps
q1_2019$started_at <- ymd_hms(q1_2019$started_at)
q1_2019$ended_at   <- ymd_hms(q1_2019$ended_at)

q1_2020$started_at <- ymd_hms(q1_2020$started_at)
q1_2020$ended_at   <- ymd_hms(q1_2020$ended_at)

# Fix ride_id mismatch
q1_2019$ride_id <- as.character(q1_2019$ride_id)
q1_2020$ride_id <- as.character(q1_2020$ride_id)

6. Align Column Order

required_cols <- c(
  "ride_id", "rideable_type",
  "started_at", "ended_at",
  "start_station_name", "start_station_id",
  "end_station_name", "end_station_id",
  "start_lat", "start_lng",
  "end_lat", "end_lng",
  "member_casual"
)

q1_2019 <- q1_2019 %>% select(all_of(required_cols))
q1_2020 <- q1_2020 %>% select(all_of(required_cols))

7. Combine Datasets Safely

df <- bind_rows(q1_2019, q1_2020)

8. Free Memory

rm(q1_2019, q1_2020)

9. Create Ride Length and Day of Week

df <- df %>%
  mutate(
    ride_length = as.numeric(difftime(ended_at, started_at, units = "mins")),
    day_of_week = wday(started_at, label = TRUE),
    month = month(started_at, label = TRUE),
    year = year(started_at)
  )

10. Summary Statistics

df %>%
  group_by(member_casual) %>%
  summarise(
    avg_length = mean(ride_length, na.rm = TRUE),
    median_length = median(ride_length, na.rm = TRUE),
    total_rides = n()
  )

# 12. Member vs Casual — Monthly Ride Length
